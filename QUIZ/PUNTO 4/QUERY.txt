SELECT SUBSTR(SEDE_CODIGO, 1, 2) AS CODIGO_DEPARTAMENTO, COUNT(DISTINCT SEDE_CODIGO) AS TOTAL_SEDES_UNICAS_CON_05
FROM ActividadesenlosquesonutilizadoslosbienesTICparaenseanza2023copia
WHERE ACTIVIDAD_CODIGO = '05'
GROUP BY SUBSTR(SEDE_CODIGO, 1, 2)
HAVING COUNT(DISTINCT SEDE_CODIGO) > 500
ORDER BY TOTAL_SEDES_UNICAS_CON_05 DESC;
